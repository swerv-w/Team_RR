# 작업 시작 전 브랜치 최신화 관리
git checkout develop
git pull origin develop
git checkout {로컬에서 자신이 작업하고 있는 브랜치명}
git merge develop              # develop 최신 내용을 내 브랜치에 반영

# 작업 완료 후 브랜치 최신화 관리
git add .
git commit -m "{자신이 담당한 기본 알고리즘명} 구현"
git push origin {로컬에서 자신이 작업하고 있는 브랜치명}
